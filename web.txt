import React, { useState } from 'react';
import { View, TextInput, Text, StyleSheet, Button, Alert } from 'react-native';

// 1. Definimos a interface completa do Usuário (como vem do Back-end)
interface Usuario {
  id: number;
  nome: string;
  email: string;
  senha: string;
  ativo: boolean;
}

// 2. Criamos o tipo do Formulário usando Omit
// O usuário não digita 'id', nem define se é 'ativo' na criação
type FormDataCadastro = Omit<Usuario, 'id' | 'ativo'>;

// 3. Definimos o estado inicial
const initialState: FormDataCadastro = {
  nome: '',
  email: '',
  senha: '',
};

export default function TelaCadastro() {
  // 4. useState tipado com o nosso FormDataCadastro
  const [formData, setFormData] = useState<FormDataCadastro>(initialState);

  // 5. Função Genérica de Mudança (O Pulo do Gato!)
  // 'field' só pode ser uma chave válida de FormDataCadastro (keyof)
  const handleChange = (field: keyof FormDataCadastro, value: string) => {
    setFormData((prev) => ({
      ...prev,       // Mantém os valores antigos
      [field]: value // Atualiza apenas o campo dinâmico
    }));
  };

  const handleSalvar = () => {
    // Validação simples
    if (!formData.nome || !formData.email) {
      Alert.alert('Erro', 'Preencha nome e email!');
      return;
    }
    console.log('Enviando para API:', formData);
    Alert.alert('Sucesso', 'Cadastro realizado!');
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Criar Conta</Text>

      <TextInput
        style={styles.input}
        placeholder="Nome"
        value={formData.nome}
        // Passamos o nome da chave 'nome' explicitamente
        onChangeText={(text) => handleChange('nome', text)} 
      />

      {/* Campo Email */}
      <TextInput
        style={styles.input}
        placeholder="Email"
        value={formData.email}
        onChangeText={(text) => handleChange('email', text)}
        keyboardType="email-address"
      />

      {/* Campo Senha */}
      <TextInput
        style={styles.input}
        placeholder="Senha"
        value={formData.senha}
        onChangeText={(text) => handleChange('senha', text)}
        secureTextEntry
      />

      <View style={styles.buttonContainer}>
        <Button title="Cadastrar" onPress={handleSalvar} />
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex: 1, justifyContent: 'center', padding: 20 },
  title: { fontSize: 24, fontWeight: 'bold', marginBottom: 20, textAlign: 'center' },
  input: { borderWidth: 1, borderColor: '#ccc', padding: 10, marginBottom: 15, borderRadius: 5 },
  buttonContainer: { marginTop: 10 },
});
